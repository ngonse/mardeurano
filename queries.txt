query Products {
  allShopifyProduct {
    nodes {
      id
      shopifyId
      title
      availableForSale
      descriptionHtml
      publishedAt
      handle
      priceRange {
        minVariantPrice {
          currencyCode
          amount
        }
        maxVariantPrice {
          currencyCode
          amount
        }
      }
      variants {
        shopifyId
        availableForSale
        selectedOptions {
          name
          value
        }
      }
      images {
        localFile {
          childImageSharp {
            fixed(width: 600, height: 800) {
              src
            }
          }
        }
      }
    }
  }
}
